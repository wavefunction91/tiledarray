82  void init_range_data(const Index1& lower_bound, const Index2& upper_bound) {
83  // Construct temp pointers
84  size_type* MADNESS_RESTRICT const lower = data_;
85  size_type* MADNESS_RESTRICT const upper = lower + rank_;
86  size_type* MADNESS_RESTRICT const extent = upper + rank_;
87  size_type* MADNESS_RESTRICT const stride = extent + rank_;
88  const auto* MADNESS_RESTRICT const lower_data = detail::data(lower_bound);
89  const auto* MADNESS_RESTRICT const upper_data = detail::data(upper_bound);
90 
91  // Set the volume seed
92  volume_ = 1ul;
93  offset_ = 0ul;
94 
95  // Compute range data
96  for (int i = int(rank_) - 1; i >= 0; --i) {
97  // Check input dimensions
98  TA_ASSERT(lower_data[i] <= upper_data[i]);
99 
100  // Compute data for element i of lower, upper, and extent
101  const size_type lower_bound_i = lower_data[i];
102  const size_type upper_bound_i = upper_data[i];
103  const size_type extent_i = upper_bound_i - lower_bound_i;
104 
105  lower[i] = lower_bound_i;
106  upper[i] = upper_bound_i;
107  extent[i] = extent_i;
108  stride[i] = volume_;
109  offset_ += lower_bound_i * volume_;
110  volume_ *= extent_i;
111  }
112  }

257  void init_range_data(
258  const Permutation& perm,
259  const size_type* MADNESS_RESTRICT const other_lower_bound,
260  const size_type* MADNESS_RESTRICT const other_upper_bound) {
261  // Create temporary pointers to this range data
262  auto* MADNESS_RESTRICT const lower = data_;
263  auto* MADNESS_RESTRICT const upper = lower + rank_;
264  auto* MADNESS_RESTRICT const extent = upper + rank_;
265  auto* MADNESS_RESTRICT const stride = extent + rank_;
266 
267  // Copy the permuted lower, upper, and extent into this range.
268  for (unsigned int i = 0u; i < rank_; ++i) {
269  const auto perm_i = perm[i];
270 
271  // Get the lower bound, upper bound, and extent from other for rank i.
272  const auto other_lower_bound_i = other_lower_bound[i];
273  const auto other_upper_bound_i = other_upper_bound[i];
274  const auto other_extent_i = other_upper_bound_i - other_lower_bound_i;
275 
276  // Store the permuted lower bound, upper bound, and extent
277  lower[perm_i] = other_lower_bound_i;
278  upper[perm_i] = other_upper_bound_i;
279  extent[perm_i] = other_extent_i;
280  }
281 
282  // Recompute stride, offset, and volume
283  volume_ = 1ul;
284  offset_ = 0ul;
285  for (int i = int(rank_) - 1; i >= 0; --i) {
286  const auto lower_i = lower[i];
287  const auto extent_i = extent[i];
288  stride[i] = volume_;
289  offset_ += lower_i * volume_;
290  volume_ *= extent_i;
291  }
292  }
